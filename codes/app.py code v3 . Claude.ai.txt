# ============================================
# SCRAPER FUNCTION (Using requests - simpler & reliable)
# ============================================
def scrape_website(url: str) -> str:
    """
    Scrapes a website and returns its full HTML content.
    Uses requests library (simple and reliable).
    
    Args:
        url: The URL of the website to scrape
        
    Returns:
        The full HTML content of the page as a string
    """
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36",
        "Accept": "text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8",
        "Accept-Language": "en-US,en;q=0.5",
        "Accept-Encoding": "gzip, deflate, br",
        "Connection": "keep-alive",
    }
    
    try:
        response = requests.get(url, headers=headers, timeout=30, verify=True)
        response.raise_for_status()  # Raise exception for bad status codes
        return response.text
    except requests.exceptions.SSLError:
        # Try without SSL verification as fallback
        response = requests.get(url, headers=headers, timeout=30, verify=False)
        response.raise_for_status()
        return response.text


def scrape_website_playwright(url: str) -> str:
    """
    Scrapes a website using Playwright (for JavaScript-heavy sites).
    Falls back to requests if Playwright fails.
    """
    try:
        from playwright.sync_api import sync_playwright
        
        with sync_playwright() as p:
            browser = p.chromium.launch(headless=True)
            context = browser.new_context(
                viewport={"width": 1920, "height": 1080},
                user_agent="Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
            )
            page = context.new_page()
            page.goto(url, wait_until="networkidle", timeout=60000)
            page.wait_for_load_state("domcontentloaded")
            time.sleep(2)
            html_content = page.content()
            context.close()
            browser.close()
            return html_content
    except Exception as e:
        raise Exception(f"Playwright error: {str(e)}")
